427  EventDataType *pDataTemp = nullptr;
428  while (m_event_generated) {
429  /* assert(m_new_state < m_max_states) */
430  if (m_new_state >= m_max_states) {

434  pDataTemp = static_cast<EventDataType *>(m_event_data);
