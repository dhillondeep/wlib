228 __Allocator *memory_get_allocator(size32_type size, bool anObject) {
229  // Based on the size, find the next higher powers of two value.
230  // Add REQUIRED_EXTRA_BUFFER to the requested block size to hold the size
231  // of an Allocator* within the block memory region. Most blocks are powers of two,
232  // however some common allocator block sizes can be explicitly defined
233  // to minimize wasted storage. This offers application specific tuning.
234  // These restrictions can only imposed if no pool is being used
235 
236  size32_type blockSize = size + REQUIRED_EXTRA_BUFFER;
237 
238  if (anObject)
239  blockSize += OBJECT_INFO_BUFFER;
240 
241 
242 #if !defined(DYNAMIC_POOL) && !defined(STATIC_POOL)
243  // set custom blocks if pool is not used
244 
245  if (blockSize > 256 && blockSize <= 396)
246  blockSize = 396;
247  else if (blockSize > 512 && blockSize <= 768)
248  blockSize = 768;
249  else
250  blockSize = nextHigher<size_t>(blockSize);
251 #endif

254 
255 #if !defined(DYNAMIC_POOL) && !defined(STATIC_POOL)
